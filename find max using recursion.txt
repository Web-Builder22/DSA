#include<iostream>
using namespace std;

void max(int arr[],int size,int index,int &maxi){
    if(size==index){
        return ;
    }
    
    if(arr[index]>maxi){
        maxi=arr[index];
    }
    max(arr,size,index+1,maxi);
    
}
int main(){
    int arr[]={1,2,3,4};
